<script>


let totals = {

  utilization: 0,

  scheduling: 0,

  staffing: 0,

  delay: 0

};


function calcUtilization() {

  let headcount = +document.getElementById("u_headcount").value;

  let rate = +document.getElementById("u_rate").value;

  let target = +document.getElementById("u_target").value / 100;

  let actual = +document.getElementById("u_actual").value / 100;


  let hours = 2080;

  let loss = headcount * hours * (target - actual) * rate;


  totals.utilization = loss;


  document.getElementById("u_result").innerText =

    "Estimated Lost Revenue: $" + loss.toLocaleString();

}


function calcScheduling() {

  let hours = +document.getElementById("s_hours").value;

  let managers = +document.getElementById("s_managers").value;

  let rate = +document.getElementById("s_rate").value;


  let weeks = 48;

  let loss = hours * weeks * managers * rate;


  totals.scheduling = loss;


  document.getElementById("s_result").innerText =

    "Estimated Lost Revenue: $" + loss.toLocaleString();

}


function calcStaffing() {

  let percent = +document.getElementById("o_percent").value / 100;

  let value = +document.getElementById("o_value").value;

  let projects = +document.getElementById("o_projects").value;

  let margin = +document.getElementById("o_margin").value / 100;


  let loss = percent * value * projects * margin;


  totals.staffing = loss;


  document.getElementById("o_result").innerText =

    "Estimated Margin Loss: $" + loss.toLocaleString();

}


function calcDelay() {

  let delay = +document.getElementById("d_delay").value;

  let value = +document.getElementById("d_value").value;

  let projects = +document.getElementById("d_projects").value;


  let loss = delay * value * projects * 0.02;


  totals.delay = loss;


  document.getElementById("d_result").innerText =

    "Estimated Delay Impact: $" + loss.toLocaleString();

}


function calcTotal() {

  let total =

    totals.utilization +

    totals.scheduling +

    totals.staffing +

    totals.delay;


  document.getElementById("total_result").innerText =

    "Total Estimated Impact: $" + total.toLocaleString();

}
